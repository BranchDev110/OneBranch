import { baseApi } from "./base";

import { CreateNewUserBody } from "@/types/user.types";

const usersApi = baseApi.injectEndpoints({
  endpoints: (build) => ({
    createUser: build.mutation<any, CreateNewUserBody>({
      queryFn: async ({ email, password }) => {
        // TO DO: CALL FIREBASE
        try {
          const result = { email, password };
          return { data: result };
        } catch (e) {
          return { error: e };
        }
      },
    }),
    getUserById: build.query<any, string>({
      queryFn: (userId) => {
        try {
          const result = { id: userId };
          return { data: result };
        } catch (e) {
          return { error: e };
        }
      },
    }),
  }),
});
export const { useGetUserByIdQuery, useCreateUserMutation } = usersApi;
export { usersApi };
